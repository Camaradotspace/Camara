## Install

```bash
npm install --save camara
```

```bash
yarn add camara
```

## Usage

```tsx
import React from 'react';
import { Button } from 'camara';

const App = () => {
  return (
    <>
      <Button>Camara</Button>
    </>
  );
};

export default App;
```
